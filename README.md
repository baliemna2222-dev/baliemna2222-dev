const INTERVAL = 5000;
let current = 0;
let timer, progTimer;

// Stars
const starsEl = document.getElementById('stars');
for (let i = 0; i < 80; i++) {
  const s = document.createElement('div');
  s.className = 'star';
  const size = Math.random() * 2.5 + 0.5;
  s.style.cssText = `width:${size}px;height:${size}px;top:${Math.random()*100}%;left:${Math.random()*100}%;--d:${(Math.random()*3+2).toFixed(1)}s;--o:${(Math.random()*0.6+0.2).toFixed(2)};animation-delay:${(Math.random()*4).toFixed(1)}s`;
  starsEl.appendChild(s);
}

// Dots
const dotsEl = document.getElementById('dots');
quotes.forEach((_, i) => {
  const d = document.createElement('div');
  d.className = 'dot' + (i === 0 ? ' active' : '');
  d.onclick = () => goTo(i);
  dotsEl.appendChild(d);
});

function updateDots() {
  document.querySelectorAll('.dot').forEach((d, i) => {
    d.classList.toggle('active', i === current);
  });
}

function showQuote(idx, dir = 1) {
  const qt = document.getElementById('qtext');
  const qa = document.getElementById('qauthor');
  qt.classList.add('fade');
  qa.classList.add('fade');
  setTimeout(() => {
    qt.textContent = quotes[idx].text;
    qa.textContent = quotes[idx].author;
    qt.classList.remove('fade');
    qa.classList.remove('fade');
    document.getElementById('counter').textContent = `${idx + 1} / ${quotes.length}`;
    updateDots();
    startProgress();
  }, 500);
}

function startProgress() {
  const bar = document.getElementById('progress');
  bar.style.transition = 'none';
  bar.style.width = '0%';
  clearTimeout(progTimer);
  setTimeout(() => {
    bar.style.transition = `width ${INTERVAL}ms linear`;
    bar.style.width = '100%';
  }, 30);
}

function goTo(idx) {
  clearInterval(timer);
  current = (idx + quotes.length) % quotes.length;
  showQuote(current);
  timer = setInterval(next, INTERVAL);
}

function next() { goTo(current + 1); }
function prev() { goTo(current - 1); }

showQuote(0);
timer = setInterval(next, INTERVAL);
